  <script>
    let cart = [];

    function addToCart(name, price) {
      const existingItem = cart.find(item => item.name === name);
      if (existingItem) {
        existingItem.qty++;
      } else {
        cart.push({ name, price, qty: 1 });
      }
      updateCart();
    }

    function updateCart() {
      const cartList = document.getElementById('cart-items');
      const cartTotal = document.getElementById('cart-total');
      cartList.innerHTML = '';

      let total = 0;
      cart.forEach((item, index) => {
        const subtotal = item.price * item.qty;
        total += subtotal;
        const li = document.createElement('li');
        li.innerHTML = `
          ${item.name} x${item.qty} - Rp${subtotal.toLocaleString('id-ID')}
          <div>
            <button onclick="changeQty(${index}, -1)" style="background:orange;color:white;border:none;border-radius:4px;padding:4px 8px;">-</button>
            <button onclick="changeQty(${index}, 1)" style="background:green;color:white;border:none;border-radius:4px;padding:4px 8px;">+</button>
            <button onclick="removeItem(${index})" style="background:red;color:white;border:none;border-radius:4px;padding:4px 8px;">x</button>
          </div>
        `;
        cartList.appendChild(li);
      });

      cartTotal.textContent = `Total: Rp${total.toLocaleString('id-ID')}`;
    }

    function changeQty(index, delta) {
      cart[index].qty += delta;
      if (cart[index].qty <= 0) cart.splice(index, 1);
      updateCart();
    }

    function removeItem(index) {
      cart.splice(index, 1);
      updateCart();
    }

    function checkout() {
      if (cart.length === 0) {
        alert("Keranjang masih kosong!");
        return;
      }

      let pesan = "Halo Warung Lezat! Saya ingin memesan:\n";
      cart.forEach(item => {
        pesan += `- ${item.name} x${item.qty} (Rp${(item.price * item.qty).toLocaleString('id-ID')})\n`;
      });
      const total = cart.reduce((sum, item) => sum + item.price * item.qty, 0);
      pesan += `\nTotal: Rp${total.toLocaleString('id-ID')}`;

      // Ganti nomor WA kasir kamu di bawah ini
      const url = `https://wa.me/6281234567890?text=${encodeURIComponent(pesan)}`;
      window.open(url, '_blank');
    }
